<?php
echo "<table border='1'>";
for ($row = 1; $row <= 8; $row++) {
    echo "<tr>";
    for ($col = 1; $col <= 8; $col++) {
        $color = ($row + $col) % 2 == 0 ? 'black' : 'white';
        echo "<td style='width:50px;height:50px;background:$color'></td>";
    }
    echo "</tr>";
}
echo "</table>";
?>
