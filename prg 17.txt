<?php
$balance = 1000;

function deposit($amount) {
    global $balance;
    $balance += $amount;
    echo "Deposited $amount. Balance: $balance<br>";
}

function withdraw($amount) {
    global $balance;
    if ($amount <= $balance) {
        $balance -= $amount;
        echo "Withdrew $amount. Balance: $balance<br>";
    } else {
        echo "Insufficient funds.<br>";
    }
}

deposit(500);
withdraw(300);
withdraw(1500);
?>
